import { Pipe, PipeTransform } from '@angular/core';
import * as _ from '../app/consts/objclassConsts.json';

@Pipe({
    name: 'objectIcon',
})
export class ObjectIcon implements PipeTransform {

    /**
     * Transforms an objclassID to an icon string.
     *
     * @param value
     * @returns {string}
     */
    transform(value: number): string {

        let newValue = '';

        switch (value) {
            // <editor-fold desc="Word">
            case _.Q_Dokument_doc:
            case _.MS_Word_Vorlage:
            case _.Systemdesign2:
            case _.Q_Dokument_docx:
            case _.MS_Word:
            case _.Systemdesign1:
            case _.MS_Word_2007:
                // </editor-fold>
                newValue = 'tie-file-word-icon';
                break;
            // <editor-fold desc="Excel">
            case _.Q_Dokument_xls:
            case _.XSLT_Stylesheet:
            case _.MS_Excel_2007:
            case _.Projektplan3:
            case _.MS_Excel_Makro:
            case _.Projektplan2:
            case _.MS_Excel:
                // </editor-fold>
                newValue = 'tie-file-excel-icon';
                break;
            // <editor-fold desc="Powerpoint">
            case _.MS_Powerpoint:
            case _.MS_Powerpoint_2007:
            case _.MS_Powerpoint_ppt:
                // </editor-fold>
                newValue = 'tie-document-file-ppt-icon';
                break;
            // <editor-fold desc="PDF">
            case _.Q_Dokument_pdf:
            case _.Portable_Document_Format:
                // </editor-fold>
                newValue = 'tie-file-pdf-icon';
                break;
            // <editor-fold desc="Dossier">
            case _.Dossier:
                // </editor-fold>
                newValue = 'tie-folder-open-icon';
                break;
            // <editor-fold desc="Register">
            case _.Register:
                // </editor-fold>
                newValue = 'tie-drawer-icon';
                break;
            // <editor-fold desc="Images">
            case _.GIF_Bilddatei:
            case _.PNG_Bild:
            case _.JPEG_Bilddatei1:
            case _.JPEG_Bilddatei2:
            case _.BMP_Bilddatei:
            case _.DXF_Bilddatei:
            case _.PCX_Bilddatei:
            case _.QuickTime_Bilddatei:
            case _.TIFF_Bilddatei:
                // </editor-fold>
                newValue = 'tie-file-picture-icon';
                break;
            // <editor-fold desc="Videos">
            case _.Video:
            case _.MPG_Audio__Video:
            case _.WMV_Video:
            case _.Film_MP4:
            case _.QuickTime_Film:
            case _.M1V_Audio__Videodatei:
            case _.MPEG_Audio__Video1:
            case _.MPEG_Audio__Video2:
                // </editor-fold>
                newValue = 'tie-file-video-icon';
                break;
            default:
                newValue = 'tie-file-empty-icon';
        }

        return newValue;
    }
}
