import { Injectable } from '@angular/core';
import { HttpInterceptorService } from './httpInterceptor';
import { ResponseContentType } from '@angular/http';
import { TIEObject } from '../app/models/TIEObject';
import { AuthService } from './auth';

@Injectable()
export class RestService {

    SET_FAVORITE: string = 'Object_AddToFavorites/';
    REMOVE_FAVORITE: string = 'Object_RemoveFromFavorites/';

    constructor(private intercepted: HttpInterceptorService,
                private auth: AuthService) {
        console.info('Enter: RestService');
        this.SET_FAVORITE = 'Object_AddToFavorites/';
        this.REMOVE_FAVORITE = 'Object_RemoveFromFavorites/';
    }

    /**
     * First request to be made. This represents the First node of the object tree.
     *
     * @returns {Promise<TIEObject>}
     */
    getEntryPoint(): Promise<TIEObject> {
        return this.intercepted.get(this.auth.endpoint + 'objects?q=Dossier-App-My-Favoriten-Sucher')
            .then(data => data.json())
            .catch(err => console.error('Error in RestService', err));
    }

    /**
     * Fetches the objclass if existent. Otherwise returns undefined in a Promise.
     *
     * @param object
     * @returns {Promise<TIEObject>}
     */
    getObjectClass(object: TIEObject): Promise<TIEObject> {
        let url = object.links.find(it => it.rel === 'objclass');

        return (url) ? this.intercepted.get(url.href)
            .then(data => data.json()) : new Promise(resolve => resolve(undefined))
            .catch(err => console.error('Error in RestService', err));
    }

    /**
     * Fetches the attribute definitions from an object.
     *
     * @param object
     * @returns {Promise<TIEObject>}
     */
    getAttributesEdit(object: TIEObject): Promise<TIEObject> {
        let url = object.links.find(it => it.rel === 'ATTRIBUTES_EDIT' || it.rel === 'BULK_ATTRIBUTES');

        return (url) ? this.intercepted.get(url.href)
            .then(data => data.json()) : new Promise(resolve => resolve(undefined))
            .catch(err => console.error('Error in RestService', err));
    }

    /**
     * Fetches an object.
     *
     * @param object
     * @returns {Promise<TIEObject>}
     */
    getObject(object: TIEObject): Promise<TIEObject> {
        let url = object.links.find(it => it.rel === 'object');

        // TODO: Hotfix only!!!
        return (url) ? this.intercepted.get(url.href)
            .then(data => data.json()) : new Promise(resolve => resolve(undefined))
            .catch(err => console.error('Error in RestService', err));
    }

    /**
     * Generalized GET request method.
     *
     * @param url
     * @param file
     * @returns {Promise<any>} Response or TIEObject
     */
    requestURLGet(url: string, file?: boolean): Promise<any> {
        if (file) {
            let attributes = {responseType: ResponseContentType.Blob};
            return this.intercepted.get(url, attributes)
                .catch(err => console.error('Error in RestService', err));
        } else {
            return this.intercepted.get(url)
                .then(data => data.json())
                .catch(err => console.error('Error in RestService', err));
        }
    }

    /**
     * Generalized PUT request method.
     *
     * @param url
     * @param attributes
     * @returns {Promise<TIEObject>}
     */
    requestURLPut(url: string, attributes?: any): Promise<TIEObject> {
        return this.intercepted.put(url, attributes)
            .then(data => data.json())
            .catch(err => console.error('Error in RestService', err));
    }

    /**
     * Returns all the favorites of the logged in user
     *
     * @param objId
     * @param activityId
     * @param parentId
     * @returns {Promise<TIEObject>}
     */
    getFavorites(objId: number, activityId: number, parentId: number): Promise<TIEObject> {
        return this.intercepted.get(this.auth.endpoint + 'Object_FindInObjlist/' +
            objId + '/' + activityId + '/' + parentId)
            .then(data => data.json())
            .catch(err => console.error('Error in RestService', err));
    }

    /**
     * Get the objclass from the uploading object
     *
     * @param uri
     * @param urlGetId DRAGDROP link of the register or dossier where we are
     * @returns {Promise<number>}
     */
    getObjclassIdFromURI(uri: string, urlGetId: string): Promise<number> {
        let resultSet = [];
        return this.requestURLGet(urlGetId)
            .then(result => {
                let options = result.objClasses;
                // Get Extension from path
                let extensionFromUrl = '.' + uri.split('.').pop();
                for (let i = 0; i < options.length; i++) {
                    if (options[i].extension === extensionFromUrl) {
                        resultSet.push(options[i].objclassId);
                    }
                }
                if (resultSet.length === 1) {
                    return resultSet[0];
                } else if (resultSet.length > 1) {
                    return resultSet[1];
                } else {
                    return 0;
                }
            });
    }

    /**
     * Setting an object as favorite.
     *
     * @param objId
     * @returns {Promise<void>}
     */
    setAsFavorite(objId: number): Promise<any> {
        console.log('Favorite TRUE for ' + objId, this.auth.endpoint, this.SET_FAVORITE);
        return this.requestURLPut(this.auth.endpoint + this.SET_FAVORITE + objId);
    }

    /**
     * Removing an object from the favorite list.
     *
     * @param objId
     * @returns {Promise<void>}
     */
    removeFromFavorite(objId: number): Promise<any> {
        console.log('Favorite FALSE for ' + objId, this.auth.endpoint);
        return this.requestURLPut(this.auth.endpoint + this.REMOVE_FAVORITE + objId);
    }
}
