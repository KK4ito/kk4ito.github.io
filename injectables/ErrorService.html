import { Injectable } from '@angular/core';
import { ToastController } from 'ionic-angular';
import { Response } from '@angular/http';
import { TranslateService } from 'ng2-translate';
import { LoadingService } from './loading';

@Injectable()
export class ErrorService {

    constructor(private toastCtrl: ToastController,
                private translate: TranslateService,
                private loader: LoadingService) {
        console.info('Enter: ErrorService');
    }

    /**
     * Handels all errors with custom messages.
     *
     * @param message
     * @param params
     */
    handleCustomError(message: string, ...params: Array<any>): void {
        console.log('Error: handleCustomError', params);
        let toast = this.toastCtrl.create({
            message: message,
            duration: 3000,
            position: 'bottom',
        });

        toast.present();
    }

    /**
     * Handels all errors concerning HTTP.
     *
     * @param errorResponse
     * @param info
     * @param params
     */
    handleHttpError(errorResponse: Response, info: string, ...params: Array<any>): void {
        console.warn('Error: handleHttpError', params);
        console.warn(info);
        console.warn(errorResponse);
        let toast = this.toastCtrl.create({
            message: this.errorResponseSwitch(errorResponse).message,
            duration: 3000,
            position: 'bottom',
        });

        toast.present();
    }

    /**
     * Handels all errors concerning logout.
     *
     * @param errorObject
     * @param params
     */
    handleLogoutError(errorObject: any, ...params: Array<any>): void {
        console.warn('Error: handleLogoutError', params);
        console.warn(errorObject);
        let toast = this.toastCtrl.create({
            message: this.translate.instant('ERROR_LOGOUT'),
            duration: 3000,
            position: 'bottom',
        });

        toast.present();
    }

    /**
     * Handels all error concerning files.
     *
     * @param errorObject
     * @param message
     * @param params
     */
    handleFileError(errorObject: any, message?: string, ...params: Array<any>): void {
        console.warn('Error: handleFileError', params);
        console.warn(errorObject);
        let toast = this.toastCtrl.create({
            message: (message) ? message : this.translate.instant('ERROR_FILE'),
            duration: 3000,
            position: 'bottom',
        });

        toast.present();
    }

    /**
     * Handels all error concerning DB.
     *
     * @param errorObject
     * @param message
     * @param params
     */
    handleDBError(errorObject: any, message?: string, ...params: Array<any>): void {
        console.warn('Error: handleDBError', params);
        console.warn(errorObject);
        if (message) console.warn(message);

        /*
        let toast = this.toastCtrl.create({
            message: (message) ? message : 'Datenbank Error',
            duration: 3000,
            position: 'bottom',
        });
        toast.present();
        */
    }

    /**
     * Switch for deciding which error the response reflects.
     *
     * @param errorResponse
     * @returns {{message: string, toast: boolean}}
     */
    private errorResponseSwitch(errorResponse: Response): {message: string, toast: boolean} {
        let status: number = errorResponse.status;
        let result: any = {
            message: '',
            toast: true,
        };
        switch (status) {
            case 404:
                result.message = this.translate.instant('ERROR_FILE');
                break;
            case 401:
                result.message = this.translate.instant('ERROR_AUTHENTIFICATION');
                this.loader.loading(false);
                break;
            default:
                result.message = this.translate.instant('ERROR_UNKNOWN');
                break;
        }

        return result;
    }
}
