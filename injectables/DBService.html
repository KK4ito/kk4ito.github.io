import { Injectable } from '@angular/core';
import { SQLite } from 'ionic-native';
import { ErrorService } from './error';
import { Platform } from 'ionic-angular';
import { DBObject, DBOfflineObject } from '../app/models/DBObject';
import { TIEObject } from '../app/models/TIEObject';
import * as CONFIG from '../assets/settings.json';

declare let document: any, window: any;

@Injectable()
export class DBService {
    db: SQLite = new SQLite();
    /**
     * Variable contains all possible actions on an object
     */

    constructor(private error: ErrorService,
                private platform: Platform) {
        console.info('Enter: DBService');

    }

    /**
     *  explicit Implementation of Object.values since not all browser supports the
     */
    getObjectValues(object: any) {
        let vals = Object.keys(object).map(function(k) {
            return object[k];
        });
        return vals;
    }
    /**
     * DB initializier. If tables already exist nothing happens.
     *
     * @returns {Promise<void>}
     */
    initDB(): Promise<void> {
        return this.platform.ready()
            // .then(() => this.db = new SQLite())
            .then(() => this.db.openDatabase({name: 'tie', location: 'default'}))
            .then(() => this.db.executeSql('CREATE TABLE IF NOT EXISTS USERS (' +
                'name text NOT NULL,' +
                'config text NOT NULL,' +
                'token text NOT NULL,' +
                'PRIMARY KEY (name))', []))
            .then(() => this.db.executeSql('CREATE TABLE IF NOT EXISTS OBJECTS (' +
                'id integer NOT NULL, ' +
                'type integer NOT NULL, ' +
                'data text NOT NULL, ' +
                'parent integer, ' +
                'active integer, ' +
                'owner text, ' +
                'PRIMARY KEY (id, parent, owner), ' +
                'UNIQUE (id,parent,owner) ON CONFLICT REPLACE ' +
                'CONSTRAINT Objects_Objects FOREIGN KEY (parent) REFERENCES Objects (id) ' +
                'ON DELETE CASCADE ' +
                'CONSTRAINT Objects_Objects2 FOREIGN KEY (owner) REFERENCES Users (name) ' +
                'ON DELETE CASCADE ' +
                'ON UPDATE CASCADE)', []))
            .then(() => this.db.executeSql('CREATE TABLE IF NOT EXISTS OFFLINEFILES (' +
                'offlineId integer NOT NULL, ' +
                'localpath text NOT NULL, ' +
                'data text NOT NULL, ' +
                'owner text NOT NULL, ' +
                'fileType text NOT NULL)', []))
            .then(() => console.log('DB created'))
            .catch(err => {
                this.error.handleDBError(err);
            });
    }

    /**
     * Clears the DB tables.
     *
     * @returns {Promise<void>}
     */
    dbClear(): Promise<void> {
        return this.platform.ready()
            .then(() => this.db.executeSql('DROP TABLE IF EXISTS OBJECTS', []))
            .then(() => this.db.executeSql('DROP TABLE IF EXISTS OFFLINEFILES', []))
            .then(() => this.db.executeSql('DROP TABLE IF EXISTS USERS', []))
            .catch(err => {
                this.error.handleDBError(err);
            });
    }

    /**
     * Returns a dump of all objects in DB.
     *
     * @returns {Promise<Array<DBObject>>}
     */
    getAllObjects(): Promise<Array<DBObject>> {
        return this.platform.ready()
            .then(() => this.db.executeSql('SELECT * FROM OBJECTS', []))
            .then(data => {
                let res: Array<any> = [];
                for (let i = 0; i < data.rows.length; i++) {
                    res[i] = data.rows.item(i);
                }
                return res;
            });
    }

    /**
     * Returns a dump of all users in DB.
     *
     * @returns {Promise<Array<string>>}
     */
    getAllUsers(): Promise<Array<string>> {
        return this.platform.ready()
            .then(() => this.db.executeSql('SELECT * FROM USERS', []))
            .then(data => {
                let res: Array<any> = [];
                for (let i = 0; i < data.rows.length; i++) {
                    res[i] = data.rows.item(i);
                }
                return res;
            });
    }

    /**
     * Multi transactional create.
     *
     * @param objects
     * @param owner
     * @param parent
     * @param active
     * @returns {Promise<void>}
     */
    createObjects(objects: Array<TIEObject>, owner: string, parent: number, active: number = 0): Promise<void> {
        let sql = 'INSERT INTO OBJECTS VALUES (?,?,?,?,?,?)';
        let params = [];
        return this.db.transaction((txn) => {
            if (active === 1) {
                let testsql = 'UPDATE OBJECTS SET active=? WHERE owner=?';
                let testParams = [
                    0,
                    owner,
                ];
                txn.executeSql(testsql, testParams, () => {},
                    (tx, error) => console.error('Update object error: ' + error.message, tx));
            }
            objects.forEach((object: TIEObject) => {
                params = [
                    object.objId,
                    (parent === 0) ? 0 : this.extractObjectClassID(object),
                    JSON.stringify(object),
                    parent,
                    active,
                    owner];
                txn.executeSql(sql, params, () => {},
                    (tx, error) => console.error('Create object error: ' + error.message, tx));
            });
        })
            .catch(err => console.error('Create error', err));
    }

    /**
     * Multi transactional create.
     *
     * @param objects
     * @returns {Promise<void>}
     */
    createOfflineFiles(objects: Array<DBOfflineObject>): Promise<void> {
        let sql = 'INSERT INTO OFFLINEFILES VALUES (?,?,?,?,?)';
        let params = [];
        return this.db.transaction((txn) => {
            objects.forEach((object: DBOfflineObject) => {
                params = [object.offlineId, object.localpath, object.data, object.owner, object.fileType];
                txn.executeSql(sql, params, () => {},
                    (tx, error) => console.error('Create offlinefile error: ' + error.message, tx));
            });
        }).catch(err => console.error('Create error', err));
    }

    /**
     * Multi transactional create.
     *
     * @param objects
     * @returns {Promise<void>}
     */
    createUsers(objects: Array<string>): Promise<void> {
        let baseConfig = JSON.stringify(CONFIG);
        console.log('baseconfig for user', baseConfig);
        let sql = 'INSERT OR IGNORE INTO USERS VALUES (?,?,?)';
        let params = [];
        return this.db.transaction((txn) => {
            objects.forEach((object: string) => {
                params = [object, baseConfig, ''];
                txn.executeSql(sql, params, (tx, res) => console.log('result of exe', tx, params, res),
                    (tx, error) => console.error('Create user error: ' + error.message, tx));
            });
        }).catch(err => console.error('Create error', err));
    }

    /**
     * Single transactional read.
     *
     * @param params {key:value},{key:value},...
     * @returns {Promise<DBObject | Array<DBObject>>}
     */
    readObjects(...params): Promise<DBObject | Array<DBObject>> {
        let sql = 'SELECT * FROM OBJECTS';
        let where = params.length > 0 ? ' WHERE ' + params
                .map(f => (<any>Object).keys(f) + '=' + this.getObjectValues(f))
                .join(' AND ') : '';
        return this.db.executeSql(sql + where, [])
            .then(result => {
                let set = [];
                for (let i = 0; i < result.rows.length; i++) {
                    set.push(result.rows.item(i));
                }
                return (set.length > 1) ? set : set[0];
            })
            .catch(err => console.error('Read error', err));
    }

    /**
     * Single transactional read.
     *
     * @param params {key:value},{key:value},...
     * @returns {Promise<DBOfflineObject | Array<DBOfflineObject>>}
     */
    readOfflineFiles(...params): Promise<DBOfflineObject | Array<DBOfflineObject>> {
        let sql = 'SELECT * FROM OFFLINEFILES';
        let where = params.length > 0 ? ' WHERE ' + params
                .map(f => (<any>Object).keys(f) + '="' + this.getObjectValues(f) + '"')
                // .map(f => (<any>Object).keys(f) + '="' + (<any>Object).values(f) + '"')
                .join(' AND ') : '';
        console.log('check sql hier', sql + where);
        return this.db.executeSql(sql + where, [])
            .then(result => {
                let set = [];
                for (let i = 0; i < result.rows.length; i++) {
                    set.push(result.rows.item(i));
                }
                return (set.length > 1) ? set : set[0];
            })
            .catch(err => console.error('Read error', err));
    }

    /**
     * Single transactional read.
     *
     * @param params {key:value},{key:value},...
     * @returns {Promise<string | Array<string>>}
     */
    readUsers(...params): Promise<string | Array<string>> {
        let sql = 'SELECT * FROM USERS';
        let where = params.length > 0 ? ' WHERE ' + params
                .map(f => (<any>Object).keys(f) + '="' + this.getObjectValues(f) + '"')
                .join(' AND ') : '';
        return this.db.executeSql(sql + where, [])
            .then(result => {
                let set = [];
                for (let i = 0; i < result.rows.length; i++) {
                    set.push(result.rows.item(i));
                }
                return (set.length > 1) ? set : set[0];
            })
            .catch(err => console.error('Read error', err));
    }

    /**
     * Multi transactional update.
     *
     * @param objects
     * @param owner
     * @param parent
     * @param active 1 for current viewing object
     * @returns {Promise<void>}
     */
    updateObjects(objects: Array<TIEObject>, owner: string, parent?: number, active: number = 0): Promise<void> {
        let sql = 'UPDATE OBJECTS SET id=?,type=?,data=?,owner=?,active=?';
        let optionalSql = (parent) ? ',parent=?' : '';
        let whereSql = ' WHERE owner=? AND id=?';
        sql = sql.concat(optionalSql, whereSql);
        let params = [];
        let where = [];
        console.log('updateObjects:sql', sql);
        console.log('updateObjects', objects, parent, active);
        return this.db.transaction((txn) => {
            if (active === 1) {
                let testsql = 'UPDATE OBJECTS SET active=? WHERE owner=?';
                let testParams = [
                    0,
                    owner,
                ];
                txn.executeSql(testsql, testParams, () => {},
                    (tx, error) => console.error('Update object error: ' + error.message, tx));
            }
            objects.forEach((object: TIEObject) => {
                params = [
                    object.objId,
                    (parent === 0) ? 0 : this.extractObjectClassID(object),
                    JSON.stringify(object),
                    owner,
                    active,
                ];
                where = [owner, object.objId];
                if (parent) params.push(parent);
                txn.executeSql(sql, params.concat(where), () => {},
                    (tx, error) => console.error('Update object error: ' + error.message, tx));
            });
        }).catch(err => console.error('Update error', err));
    }

    /**
     * Set new config in DB.
     *
     * @param name
     * @param config
     * @returns {Promise<void>}
     */
    updateUser(name: string, config: any): Promise<void> {
        return this.db.executeSql('UPDATE USERS SET config = ? WHERE name = ?', [JSON.stringify(config), name])
            .catch(err => console.error('Update error', err));
    }

    /**
     * Save fingerprint token for the user
     *
     * @param name
     * @param token
     * @returns {Promise<void>}
     */
    storeUserFingerPrint(name: string, token: string): Promise<void> {
        return this.db.executeSql('UPDATE USERS SET token = ? WHERE name = ?', [token, name])
            .catch(err => console.error('Save token error', err));
    }

    /**
     * Multi transactional delete.
     *
     * @param objects
     * @param owner
     * @returns {Promise<void>}
     */
    deleteObjects(objects: Array<TIEObject>, owner: string): Promise<void> {
        let sql = 'DELETE FROM OBJECTS WHERE id=? AND owner=?';
        let params = [];
        return this.db.transaction((txn) => {
            objects.forEach((object: TIEObject) => {
                params = [object.objId, owner];
                txn.executeSql(sql, params, () => {},
                    (tx, error) => console.error('Delete user error: ' + error.message, tx));
            });
        }).catch(err => console.error('Delete error', err));
    }

    /**
     * Multi transactional delete.
     *
     * @param objects
     * @returns {Promise<void>}
     */
    deleteOfflineFiles(objects: Array<DBOfflineObject>): Promise<void> {
        let sql = 'DELETE FROM OFFLINEFILES WHERE offlineId=?';
        let params = [];
        return this.db.transaction((txn) => {
            objects.forEach((object: DBOfflineObject) => {
                params = [object.offlineId];
                txn.executeSql(sql, params, () => {},
                    (tx, error) => console.error('Delete offlinefiles error: ' + error.message, tx));
            });
        }).catch(err => console.error('Delete error', err));
    }

    /**
     * Multi transactional delete.
     *
     * @param objects
     * @returns {Promise<void>}
     */
    deleteUsers(objects: Array<string>): Promise<void> {
        let sql = 'DELETE FROM USERS WHERE name=?';
        let params = [];
        return this.db.transaction((txn) => {
            objects.forEach((object: string) => {
                params = [object];
                txn.executeSql(sql, params, () => {},
                    (tx, error) => console.error('Delete user error: ' + error.message, tx));
            });
        }).catch(err => console.error('Delete error', err));
    }

    /**
     * Extracts last part of url objclass.
     *
     * @param object
     * @returns {number}
     */
    extractObjectClassID(object: TIEObject): number {
        return parseInt(object.links.find(link => link.rel === 'objclass').href.split('/').pop(), 10);
    }

    getUserConfig(name: string): Promise<Array<any>> {
        return new Promise((resolve, reject) => {
            this.db.executeSql('SELECT config FROM USERS WHERE name = ?', [name])
                .then(data => {
                    console.log('getUserConfig function', name, data.rows.item(0));
                    if (data.rows.length > 0) {
                        resolve(JSON.parse(data.rows.item(0).config));
                    } else { reject('No user found by name: ' + name); }
                })
                .catch(err => {
                    console.error('Error in DBService', err);
                    this.error.handleDBError(err);
                });
        });
    }

    rgbToHex(rgb: string): string {
        let a = rgb.split('(')[1].split(')')[0];
        let b = a.split(',');
        let c = b.map(function(x) {
            let y = parseInt(x, 10).toString(16);
            return (y.length === 1) ? '0' + y : y;
        });
        let d = '#' + c.join('');

        return d;
    }
