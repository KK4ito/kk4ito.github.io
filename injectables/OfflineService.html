import { Injectable } from '@angular/core';
import { RestService } from './rest';
import { FileService } from './file';
import { File } from 'ionic-native';
import { DBOfflineObject } from '../app/models/DBObject';
import { ToastController } from 'ionic-angular';
import { AuthService } from './auth';
import { PersistenceService } from './persistence';
import { TIEObject } from '../app/models/TIEObject';

@Injectable()
export class OfflineService {

    constructor(private persistence: PersistenceService,
                private rest: RestService,
                private toastCtrl: ToastController,
                private fileHandler: FileService,
                private auth: AuthService) {
        console.info('Enter: OfflineService');
    }

    /**
     * Saves the checked out file.
     *
     * @param localPath
     * @param owner
     * @param checkInLink
     * @param fileType
     */
    saveReference(offlineId: number, localPath: string, data: TIEObject, owner: string, fileType: string): void {
        console.log('entering saveRefernce in offline', owner, localPath, fileType, offlineId);
        this.persistence.saveReference(offlineId, localPath, data, owner, fileType);
    }

    /**
     * Returns all checked out files.
     *
     * @returns {Promise<Array<DBOfflineObject>>}
     */
    getOfflineFiles(): Promise<Array<DBOfflineObject>> {
        return this.persistence.getOfflineReference()
            .then(result => {
                console.log('result of getOfflineFile', result);
                return result;
            });
    }

    /**
     * Checks in the offline file selected.
     *
     * @param item
     */
    checkInOfflineFiles(item: DBOfflineObject): void {
        // let fileName: string = obj.attributes.objName + obj.attributes.objclassDescr;
        let fileName: string = item.localpath.split('/').pop();
        File.readAsArrayBuffer(encodeURI(
            this.fileHandler.directory + this.auth.username.value + '/' + 'edit/'), fileName)
            .then(dataInput => {
                // TODO fix with better implementation
                console.log('checkInOfflienFiles', dataInput);
                this.persistence.getOfflineReference(item.offlineId)
                    .then((dbItemById: DBOfflineObject) => {
                        console.log(dbItemById);
                        let parsedLinks = JSON.parse(dbItemById.data).links;
                        this.rest.requestURLPut(parsedLinks.find(it => it.rel === 'CHECKIN').href, dataInput)
                            .then(data => {
                                console.log(data);
                                this.fileHandler.getFile(fileName, 'edit')
                                    .then(file => file.remove(() => {
                                        this.persistence.getOfflineReference(item.offlineId)
                                            .then((dbEntry: DBOfflineObject) => {
                                                this.persistence.deleteByCheckIn([dbEntry]);
                                                // this.persistence.update('OBJECTS', data, dbEntry[0].parent);
                                            });
                                        console.log('success deleting file during checkIn process');
                                        let toast = this.toastCtrl.create({
                                            message: 'Erfolgreich Datei eingecheckt',
                                            duration: 3000,
                                            position: 'bottom',
                                        });
                                        toast.present();
                                    }));
                            });
                    });
            });
    }
}
