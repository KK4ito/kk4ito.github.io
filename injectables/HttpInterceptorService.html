import { Injectable } from '@angular/core';
import { Http, Headers, Response } from '@angular/http';
import { Storage } from '@ionic/storage';

@Injectable()
export class HttpInterceptorService {

    constructor(private http: Http,
                private storage: Storage) {
        console.info('Enter: HttpInterceptorService');
    }

    /**
     * HTTP get method helper.
     *
     * @param url
     * @param attributes
     * @returns {Promise<Response>}
     */
    get(url: string, attributes?: any): Promise<Response> {
        return new Promise((resolve, reject) => {
            let headers: Headers = new Headers();
            let options = attributes ? attributes : {};
            this.createAuthorizationHeader(headers).then(newHeaders => {
                options.headers = newHeaders;
                return this.http.get(url, options).subscribe(data => {
                    resolve(data);
                }, err => {
                    reject(err);
                });
            });
        });
    }

    /**
     * HTTP put method helper.
     *
     * @param url
     * @param attributes
     * @returns {Promise<Response>}
     */
    put(url: string, attributes?: any): Promise<Response> {
        return new Promise((resolve, reject) => {
            let headers: Headers = new Headers();
            this.createAuthorizationHeader(headers).then(newHeaders => {
                return this.http.put(url, (attributes) ? attributes : {}, {
                    headers: newHeaders,
                }).subscribe(data => {
                    resolve(data);
                }, err => {
                    reject(err);
                });
            });
        });
    }

    /**
     * HTTP post method helper.
     *
     * @param url
     * @param data
     * @returns {Promise<Response>}
     */
    post(url: string, data: any): Promise<Response> {

        return new Promise((resolve, reject) => {

            let headers: Headers = new Headers({
                'Content-Type': 'multipart/form-data',
            });
            this.createAuthorizationHeader(headers).then(newHeaders => {
                return this.http.post(url, data, {
                    headers: newHeaders,
                }).subscribe(output => {
                    resolve(output);
                }, err => {
                    reject(err);
                });
            });

        });
    }

    /**
     * Interceptor for GET, PUT and POST methods
     *
     * @param oldHeaders
     * @returns {Promise<Headers>}
     */
    createAuthorizationHeader(oldHeaders: Headers): Promise<Headers> {
        let headers = oldHeaders;
        return new Promise(resolve => {
            this.storage.get('username')
                .then( username => {
                    this.storage.get('password')
                        .then( password => {
                            headers.append('Authorization', 'Basic ' +
                                btoa(username + ':' + password));
                            resolve(headers);
                        });
                });
        });
    }
}
