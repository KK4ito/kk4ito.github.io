import { Injectable } from '@angular/core';
import { App, Platform } from 'ionic-angular';
import { File, SocialSharing } from 'ionic-native';
import { FileService } from './file';
import { OfflineService } from './offline';
import { ErrorService } from './error';
import { RestService } from './rest';
import { AuthService } from './auth';
import * as _ from '../app/consts/objclassConsts.json';
import { TIEObject } from '../app/models/TIEObject';
import { ToastController } from 'ionic-angular';
import { PdfViewerPage } from '../pages/inAppView/pdfViewer';
import { ImageViewerPage } from '../pages/imageView/imageViewer';
import { LoadingService } from './loading';
import { VideoViewerPage } from '../pages/inAppView/videoViewer';
import { TranslateService } from 'ng2-translate';
import { PersistenceService } from './persistence';

@Injectable()
export class OperationService {
    constructor(private rest: RestService,
                private fileHandler: FileService,
                private error: ErrorService,
                private platform: Platform,
                private toastCtrl: ToastController,
                private auth: AuthService,
                private persistence: PersistenceService,
                private offline: OfflineService,
                private loading: LoadingService,
                private appCtrl: App,
                private translate: TranslateService) {
        console.info('Enter: OperationService');
    }

    deleteOfflineCopy(offlineId: number): Promise<boolean> {
        return this.persistence.deleteByCheckIn([{offlineId: offlineId}])
            .then(() => true)
            .catch(() => false);
    }
    getAvailableOperations(object: TIEObject) {
        return this.persistence.filterLinks(object);
    }

    /**
     * Shares a file if we have a view link available on the object
     *
     * @returns {Promise<void>}
     */
    generalShare(item: TIEObject): Promise<void> {
        let object = Object.assign({}, item);
        let url = object.links.find(it => it.rel === 'VIEW').href;
        if (url) {
            return this.fileHandler.saveTemp(url, 'share')
                .then(data => SocialSharing.shareWithOptions({
                    message: object.attributes.descr, // not supported on some apps (Facebook, Instagram)
                    subject: object.attributes.objName, // fi. for email
                    files: [data.fullPath], // an array of filenames either locally or remotely
                    url: url,
                    chooserTitle: 'Pick an app', // Android only
                }));
        } else {
            console.error('The object got no VIEW link');
            return Promise.reject('The object got no VIEW link');
        }
    }

    markOffline(item: TIEObject): void {
        console.log('markOffline triggered for', item);
        let object = Object.assign({}, item);
        let link = object.links.find(it => it.rel === 'VIEW').href;
        if (!link) {
            console.error('The object got no VIEW link');
        } else {
            this.rest.requestURLGet(link, true)
                .then(data => {
                    let fileType: string = data.headers.get('Content-Type');
                    this.rest.requestURLGet(object.links.find(it => it.rel === 'object').href)
                        .then(obj => {
                            return this.fileHandler.saveOnCheckout(data.blob(), 'edit',
                                obj.attributes.objName + obj.attributes.objclassDescr)
                                .then(checkedOut => {
                                    this.offline.saveReference(
                                        obj.objId, checkedOut, obj,  this.auth.username.value, data.blob().type);
                                    this.fileHandler.open(checkedOut, fileType);
                                });
                        });
                })
                .catch(err => this.error.handleHttpError(err, 'OperationsModal:checkout'));
        }
    }

    attributesView(link: string): Promise<{object: TIEObject}> {
        return this.rest.requestURLGet(link, false);
    }

    attributesEdit(link: string, item: TIEObject): Promise<{object: TIEObject, urlPut: string}> {
        let returnObject = { object: null, action: '', urlPut: '' };
        return this.rest.requestURLGet(link, false)
            .then(definitions => {
                let attributeDefinition = definitions.attributeShows;
                for (let at in item.attributes) {
                    if (at !== null && attributeDefinition.hasOwnProperty(at)) {
                        attributeDefinition[at].value = item.attributes[at];
                    }
                }
                returnObject.object = attributeDefinition;
                returnObject.urlPut = link;
                return returnObject;
            });
    }

    /**
     * View a file
     *
     * @param link
     * @param object
     */
    view(link: string, object: TIEObject): void {
        this.loading.loading(true);
        let type = this.persistence.extractObjectClassID(object);
        switch (type) {
            case _.Portable_Document_Format:
            case _.Q_Dokument_pdf:
                this.fileHandler.saveTemp(link, 'view')
                    .then(data => {
                        this.loading.loading(false);
                        this.appCtrl.getActiveNav().push(PdfViewerPage, [data, object]);
                    });
                break;
            case _.GIF_Bilddatei:
            case _.PNG_Bild:
            case _.JPEG_Bilddatei1:
            case _.JPEG_Bilddatei2:
            case _.BMP_Bilddatei:
            case _.DXF_Bilddatei:
            case _.PCX_Bilddatei:
            case _.QuickTime_Bilddatei:
            case _.TIFF_Bilddatei:
                this.fileHandler.saveTemp(link, 'view')
                    .then(data => {
                        this.loading.loading(false);
                        this.appCtrl.getActiveNav().push(ImageViewerPage, data);
                    });
                break;
            case _.Video:
            case _.MPG_Audio__Video:
            case _.WMV_Video:
            case _.Film_MP4:
            case _.QuickTime_Film:
            case _.M1V_Audio__Videodatei:
            case _.MPEG_Audio__Video1:
            case _.MPEG_Audio__Video2:
                // Temporary workaround, because streaming is not working for ios
                if (this.platform.is('ios')) {
                    this.fileHandler.saveTemp(link, 'view')
                        .then(data => {
                            this.appCtrl.getActiveNav().push(VideoViewerPage, [data, object]);
                            this.loading.loading(false);
                        });
                } else {
                    console.log('before pushing video page', link, object);
                    this.loading.loading(false);
                    this.appCtrl.getActiveNav().push(VideoViewerPage, object);
                }
                break;
            default:
                this.fileHandler.saveTemp(link, 'view')
                    .then(data => {
                        this.loading.loading(false);
                        console.log('VIEW:Default', link, object, data);
                        this.fileHandler.open(data.fullPath, data.fileType);
                    });
                break;
        }
    }

    /**
     * Checks out a file
     *
     * @param link
     */
    checkout(link: string, item: TIEObject): void {
        let object = Object.assign({}, item);
        this.loading.loading(true);
        console.log('entering checkout');
        let tempCheckin = 'placeholder';
        this.rest.requestURLGet(link, true)
            .then(data => {
                let fileType: string = data.headers.get('Content-Type');
                this.rest.requestURLGet(object.links.find(it => it.rel === 'object').href)
                    .then(obj => {
                        tempCheckin = obj.links.find(it => it.rel === 'CHECKIN').href;
                        return this.fileHandler.saveOnCheckout(data.blob(), 'edit',
                            obj.attributes.objName + obj.attributes.objclassDescr)
                            .then(checkedOut => {
                                this.offline.saveReference(
                                    obj.objId, checkedOut, obj, this.auth.username.value, data.blob().type);
                                this.fileHandler.open(checkedOut, fileType);
                                this.persistence.updateSingleObjectInCache(obj, this.persistence.current.value.id);
                                this.loading.loading(false);
                            });
                    });
            })
            .catch(err => this.error.handleHttpError(err, 'OperationsModal:checkout'));
    }

    /**
     * Checks in a file
     *
     * @param link
     */
    checkIn(link: string, item: TIEObject): void {
        let object = Object.assign({}, item);
        this.rest.requestURLGet(object.links.find(it => it.rel === 'object').href)
            .then(obj => {
                let fileName: string = obj.attributes.objName + obj.attributes.objclassDescr;
                File.readAsArrayBuffer(
                    encodeURI(this.fileHandler.directory + this.auth.username.value + '/' + 'edit/'), fileName)
                    .then(dataInput => {
                        console.log(dataInput);
                        this.rest.requestURLPut(link, dataInput)
                            .then((data: TIEObject) => {
                                console.log(data);
                                this.persistence.updateSingleObjectInCache(data);
                                this.deleteOfflineCopy(data.objId);
                                this.fileHandler.getFile(fileName, 'edit')
                                    .then(file => file.remove(() => {
                                        console.log('success deleting file during checkIn process');
                                        let toast = this.toastCtrl.create({
                                            message: this.translate.instant('TOAST.CHECKIN_SUCCESS'),
                                            duration: 3000,
                                            position: 'bottom',
                                        });
                                        toast.present();
                                    }));
                            });
                    })
                    .catch(err => console.error('Error in OperationService', err));
            })
            .catch(err => this.error.handleCustomError(err, 'OperationsModal:checkIn'));
    }

    /**
     * Using the LIST link to list all items of a dossier or register
     *
     * @param link The LIST function has to be here
     * @returns {Promise<any>}
     */
    list(link: string): Promise<TIEObject> {
        console.error('Not implemented method', link);
        return null;
    }

    confirm(link: string): Promise<void> {
        console.log('entering CONFIRM operation action');
        return this.rest.requestURLPut(link)
            .then( result => console.log(result));
    }

    cancel(link: string): Promise<void> {
        console.log('entering CONFIRM operation action');
        return this.rest.requestURLGet(link)
            .then( result => console.log(result));

    }
}
