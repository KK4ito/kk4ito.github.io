import { Injectable } from '@angular/core';
import {
    File, FileOpener, DirectoryEntry, FileEntry, Flags, FileUploadResult,
    FileUploadOptions, Transfer, FilePath,
} from 'ionic-native';
import { RestService } from './rest';
import { ErrorService } from './error';
import { Response } from '@angular/http';
import { Platform, ToastController } from 'ionic-angular';
import { AuthService } from './auth';
import { DBService } from './db';
import { TIEObject } from '../app/models/TIEObject';
import { LoadingService } from './loading';
import { TranslateService } from 'ng2-translate';

declare let cordova: any;
declare let window: any;

@Injectable()
export class FileService {

    /**
     * Base folder which is readable and writable.
     */
    directory: string;
    owner: string;

    constructor(private rest: RestService,
                private error: ErrorService,
                private auth: AuthService,
                private db: DBService,
                private platform: Platform,
                private toastCtrl: ToastController,
                private load: LoadingService,
                private translate: TranslateService) {
        console.info('Enter: FileService');
        this.platform.ready()
            .then(() => {
                if (this.platform.is('android')) {
                    this.directory = File.externalApplicationStorageDirectory;
                } else if (this.platform.is('ios')) {
                    this.directory = File.documentsDirectory;
                    this.createDirectory('edit');
                } else {
                    this.directory = ''; // TODO
                }
            });
    }

    /**
     * Returns a FileEntry object with the informations about the requested file.
     *
     * @param fileName
     * @param folder
     * @returns {Promise<FileEntry>}
     */
    getFile(fileName: string, folder: string = ''): Promise<FileEntry> {
        return File.resolveDirectoryUrl(this.directory + this.auth.username.value + '/' + folder)
            .then(directory => File.getFile(directory, fileName, {create: false, exclusive: false}))
            .catch(err => {
                console.error('Error in FileService', err);
                this.error.handleFileError(err);
            });
    }

    /**
     * Saves a file to the defined folder. The folder holds multiple files at a time.
     *
     * @param blob
     * @param folder
     * @param fileName
     * @returns {Promise<string>} fullPath
     */
    saveOnCheckout(blob: Blob, folder: string, fileName: string): Promise<string> {
        let savingDirectory: DirectoryEntry;
        let fullPath: string;
        return this.getDirectory(folder)
            .then(directoryEntry => {
                savingDirectory = (<DirectoryEntry>directoryEntry);
                console.groupCollapsed('Owner and Directory');
                console.log(this.directory);
                console.groupEnd();
                fullPath = savingDirectory.nativeURL + fileName;
                // fullPath = savingDirectory.nativeURL + fileName;
            })
            // .then(() => File.writeFile(savingDirectory.nativeURL, fileName, blob, {replace: true} ))
            .then(() => File.writeFile(savingDirectory.nativeURL, fileName, blob, {replace: true} ))
            .then(() => fullPath)
            .catch(err => {
                console.error('Error in FileService', err);
                this.error.handleFileError(err, 'Checkout ist fehlgeschlagen');
            });
    }

    /**
     * Saves a file to the defined folder. The folder holds only one file at a time.
     *
     * @param uri
     * @param folder
     * @returns {Promise<{fullPath: string, fileType: string}>}
     */
    saveTemp(uri: string, folder: string): Promise<{fullPath: string, fileType: string}> {
        let response: Response;
        let savingDirectory: DirectoryEntry;
        let fileType: string;
        let blob: Blob;
        let fileName: string;
        let fullPath: string;
        return Promise.all<Response, DirectoryEntry>([
            this.rest.requestURLGet(uri, true),
            this.clearDirectory(folder)
                .then(() => this.createDirectory(folder))])
            .then(result => {
                response = result[0];
                console.log('saveTemp', result[1]);
                savingDirectory = result[1];
                fileType = response.headers.get('Content-Type');
                blob = response.blob();
                fileName = (response.headers.get('Content-Disposition'))
                    .match(/filename="(.+)"/)[1];
                fullPath = savingDirectory.nativeURL + fileName;
            })
            .then(() => File.writeFile(savingDirectory.nativeURL, fileName, blob, true))
            .then(() => ({fullPath: fullPath, fileType: fileType}))
            .catch(err => {
                console.error('Error in FileService', err);
                this.error.handleFileError(err, 'Dateidownload ist fehlgeschlagen');
            });
    }

    /**
     * Opens a file.
     *
     * @param fullPath
     * @param fileType e.g application/pdf
     */
    open(fullPath: string, fileType: string): void {
        if (this.platform.is('ios')) {
            File.resolveLocalFilesystemUrl(fullPath)
                .then(entry => {
                    cordova.InAppBrowser.open(encodeURI(entry.toURL()), '_system', 'location=no');
                    FileOpener.open(encodeURI(entry.toURL()), fileType);
                })
                .catch(err => {
                    console.error('Error in FileService', err);
                    this.error.handleFileError(err, 'Datei konnte nicht geöffnet werden');
                });
        } else {
            FileOpener.open(fullPath, fileType)
                .catch(err => {
                    console.error('Error in FileService', err);
                    this.error.handleFileError(err, 'Datei konnte nicht geöffnet werden');
                });
        }
    }

    /**
     * Creates the users' specific folder.
     */
    initUserFolder(): Promise<DirectoryEntry> {
        return File.createDir(this.directory, this.auth.username.value, false)
            .catch(err => console.error('FileService: Folder exists for user', err));
    }

    /**
     * Writes the DB dump to a json file, located in external app folder.
     *
     * @returns {Promise<string>}
     */
    writeDBToFile(): Promise<string> {
        let blob: Blob;
        let name: string = 'db_' + new Date().getTime() / 1000;
        return this.db.getAllObjects()
            .then(data => {
                blob = new Blob([JSON.stringify(data, null, 2)], { type : 'application/json' });
            })
            .then(() => File.writeFile(this.directory,
                name,
                blob,
                {replace: false}))
            .then(() => this.db.getAllUsers())
            .then(data => {
                blob = new Blob([JSON.stringify(data, null, 2)], { type : 'application/json' });
            })
            .then(() => File.writeFile(this.directory,
                name,
                blob,
                {replace: false, append: true}))
            .then(() => this.directory + name)
            .catch(err => {
                console.error('Error in FileService', err);
                this.error.handleFileError(err);
            });
    }

    /**
     * Processing the upload of a file or a picture taken.
     *
     * @param data: {objName: string, descr: string}
     * @param filePath
     * @param locationToUpload
     */
    uploadFileWithDialogProcess(data: {objName: string, descr: string}, filePath: string, locationToUpload: string): Promise<TIEObject> {
        return FilePath.resolveNativePath(filePath)
            .then(finalPath => {
                let fileName = finalPath.substr(finalPath.lastIndexOf('/') + 1,
                    finalPath.length);

                console.groupCollapsed('uploadFileWithDialogProcess');
                console.log('FINALPATH', finalPath);
                console.log('FILENAME', fileName);
                console.groupEnd();

                return this.rest.getObjclassIdFromURI(finalPath, locationToUpload)
                    .then(objclassId => {
                        let params = {
                            objclassId: objclassId,
                            attrs: '{"attributes": {   "objName": "'
                            + data.objName + '",  "descr": "' +  data.descr + '" } }',
                        };

                        let transfer = new Transfer();
                        let options: FileUploadOptions = {
                            fileKey: 'file',
                            fileName: fileName,
                            params: params,
                            chunkedMode: false,
                            headers: {
                                Connection: 'close',
                            },
                        };
                        this.load.loading(true, 'Datei wird hochgeladen');
                        return transfer.upload(finalPath, locationToUpload, options, true)
                            .then((result: FileUploadResult) => {
                                // Autoupdate View with the new item
                                let newItem: TIEObject = JSON.parse(result.response);
                                newItem.attributes.objclassID = Number(objclassId);

                                let toast = this.toastCtrl.create({
                                    message: this.translate.instant('TOAST.UPLOAD_SUCCESS'),
                                    duration: 3000,
                                    position: 'bottom',
                                });
                                toast.present();
                                this.load.loading(false);
                                return newItem;

                            })
                            .catch(err => {
                                this.load.loading(false);
                                console.error('Error in FileService', err);
                                this.error.handleFileError(err, 'Fehler bei der Dateiübertragung');
                            });
                    });
            })
            .catch(err => {
                console.error('Error in FileService', err);
                this.error.handleFileError(err);
            });
    }

    /**
     * Removes a folder and its content.
     *
     * @param folder
     * @returns {Promise<boolean>}
     */
    private clearDirectory(folder: string): Promise<boolean> {
        return File.resolveDirectoryUrl(this.directory + this.auth.username.value + '/')
            .then(fileSystem => this.onFileSystemDirSuccess(fileSystem, folder))
            .then(() => true)
            .catch(err => {
                console.error('Error in FileService', err);
                this.error.handleFileError(err);
            });
    }

    /**
     * Recreate the temp folder.
     *
     * @param folder
     * @returns {Promise<DirectoryEntry>} shows toast on error
     */
    private createDirectory(folder: string): Promise<DirectoryEntry> {
        console.log('createDirectory exists');
        return File.createDir(this.directory + this.auth.username.value + '/', folder, true)
            .catch(err => {
                console.error('Error in FileService', err);
                this.error.handleFileError(err);
            });
    }

    /**
     * Returns the requested directory.
     *
     * @param folder
     * @returns {Promise<void|DirectoryEntry>|Promise<DirectoryEntry>}
     */
    private getDirectory(folder: string): Promise<DirectoryEntry> {
        console.log('getDirectory', this.directory + this.auth.username.value + '/' +  folder);
        let options: Flags = {
            create: true,
        };
        return File.resolveDirectoryUrl(this.directory + this.auth.username.value)
            .then( (fse) => {
                return File.getDirectory(fse, folder, options);
            })
            .catch(err => {
                console.error('Error in FileService', err);
                this.error.handleFileError(err);
            });
    }

    /**
     * Deletes a folder recursively.
     *
     * @param fileSystem
     * @param folder
     * @returns {Promise<boolean>}
     */
    private onFileSystemDirSuccess(fileSystem: DirectoryEntry, folder: string): Promise<boolean> {
        let entry = fileSystem;
        return new Promise((resolve, reject) => {
            entry.getDirectory(folder, {
                create: true,
                exclusive: false,
            }, function(entry2) {
                entry2.removeRecursively(function() {
                    resolve(true);
                }, function (err) {
                    this.error.handleFileError(err);
                    reject(false);
                });
            }, function (err) {
                this.error.handleFileError(err);
                reject(false);
            });
        });
    }
}
