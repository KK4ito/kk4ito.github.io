import { Injectable } from '@angular/core';
import { PersistenceService } from './persistence';
import { TouchID } from '@ionic-native/touch-id';
import { AndroidFingerprintAuth } from '@ionic-native/android-fingerprint-auth';
import { AlertController, Platform } from 'ionic-angular';
import { Storage } from '@ionic/storage';
import { BehaviorSubject } from 'rxjs';
import { TranslateService } from 'ng2-translate';

@Injectable()
export class FingerPrintService {
    isAvailable: BehaviorSubject<boolean> = new BehaviorSubject<boolean>(false);

    constructor(private persistence: PersistenceService,
                private storage: Storage,
                private touchId: TouchID,
                private alertCtrl: AlertController,
                private translate: TranslateService,
                private androidTouchId: AndroidFingerprintAuth,
                private platform: Platform) {
        console.log('Entering Fingerprint service');
        this.platform.ready().then( () => {
            if (this.platform.is('android')) {
                console.log('Entering Fingerprint service: Android');
                this.androidTouchId.isAvailable()
                    .then(
                        () => {
                            console.log('Android FingerPrint is available!');
                            this.isAvailable.next(true);
                        },
                        err => {
                            console.error('Android FingerPrint is not available', err);
                            this.isAvailable.next(false);
                        },
                    );

            } else if (this.platform.is('ios')) {
                console.log('Entering Fingerprint service: iOS');
                this.touchId.isAvailable()
                    .then(
                        () => {
                            console.log('TouchID is available!');
                            // this.isAvailable.next(true);
                            this.isAvailable.next(false);
                        },
                        err => {
                            console.error('TouchID is not available', err);
                            this.isAvailable.next(false);
                        },
                    );
            }
        });
    }

    /**
     * Delete the fingerprint for the given username
     * @param {string} username
     */
    deleteFingerprint(username: string) {
        this.persistence.storeFingerPrintToken(username, '');
    }

    /**
     * Store the fingerprint for the given username
     * @param {string} username
     */
    storeFingerprint(username: string) {
        this.storage.get('password')
            .then(password => {
                this.platform.ready().then( () => {
                    if (this.platform.is('android')) {
                        this.androidTouchId.encrypt({
                            disableBackup: true,
                            clientId: 'ch.tie.dossierapp',
                            username: username,
                            password: password,
                        })
                            .then(result =>  {
                                this.persistence.storeFingerPrintToken(username, result.token);
                            })
                            .catch( err => console.log(' verifyFingerPrintService', err));
                    } else if (this.platform.is('ios')) {
                        console.log('STORE fngerprint with IOS touchid');
                    }
                });
            });
    }

    /**
     * Verify the fingerprint for the given username
     * @param {string} username
     * @returns {Promise<any>}
     */
    verifyFingerPrint(username: string): Promise<any> {
        return new Promise( (resolve, reject) => {
            this.platform.ready().then( () => {
                if (username) {
                    this.persistence.getUserFromDB(username)
                        .then((userObj: any) => {
                            if (userObj && userObj.token !== '') {
                                if (this.platform.is('android')) {
                                    this.androidTouchId.decrypt({
                                        disableBackup: true,
                                        clientId: 'ch.tie.dossierapp',
                                        username: username,
                                        token: userObj.token,
                                    })
                                        .then(result => {
                                            resolve(result.password);
                                        })
                                        .catch( err => {
                                            reject(err);
                                        });
                                } else if (this.platform.is('ios')) {
                                    this.touchId.verifyFingerprint('Scan your fingerprint please')
                                        .then(
                                            // TODO touchId
                                            res => {
                                                console.log('Fingerprint success', res);
                                                resolve();
                                            },
                                            err => {
                                                console.log('Error', err);
                                                reject();
                                            },
                                        );
                                }
                            } else {
                                this.alertCtrl.create({
                                    title: this.translate.instant('FINGERPRINT.NOT_FOUND.TITLE'),
                                    message: this.translate.instant('FINGERPRINT.NOT_FOUND.MESSAGE') + ' ' + username,
                                    buttons: [{
                                        text: 'Ok',
                                        role: 'cancel',
                                        handler: () => reject(),
                                    },
                                    ],
                                }).present();
                            }
                        });
                } else {
                    this.alertCtrl.create({
                        title: this.translate.instant('FINGERPRINT.USERNAME_MISSING.TITLE'),
                        message: this.translate.instant('FINGERPRINT.USERNAME_MISSING.MESSAGE'),
                        buttons: [{
                            text: 'Ok',
                            role: 'cancel',
                            handler: () => reject(),
                        },
                        ],
                    }).present();
                }
            });
        });
    }
}
