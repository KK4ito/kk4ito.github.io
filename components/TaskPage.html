import { Component } from '@angular/core';
import { MenuController, ModalController, NavController, ToastController } from 'ionic-angular';
import { RestService } from '../../services/rest';
import { AuthService } from '../../services/auth';
import { TIEObject } from '../../app/models/TIEObject';
import { TranslateService } from 'ng2-translate';
import { OperationsModal } from '../../modals/operations';
import { PersistenceService } from '../../services/persistence';
import { Object2FramesPage } from './object2Frames';
import { TaskService } from '../../services/task';

@Component({
    selector: 'ib-page-overview',
    templateUrl: 'task.html',
})
export class TaskPage {
    constructor(private auth: AuthService,
                private menuCtrl: MenuController,
                private toastCtrl: ToastController,
                private rest: RestService,
                private modalCtrl: ModalController,
                private translate: TranslateService,
                private persistence: PersistenceService,
                private navCtrl: NavController,
                private task: TaskService) {
        console.log('Enter: TaskPage');
    }

    /**
     * Check if we are logged in to enter this component
     *
     * @returns boolean
     */
    ionViewCanEnter(): boolean {
        return this.auth.authenticatedImmediate();
    }

    ionViewWillEnter(): void {
        this.initTaskList();
    }

    /**
     * Enable/ Disable sideMenus
     */
    ionViewDidEnter(): void {
        this.menuCtrl.enable(true, 'mainMenu');
    }

    initTaskList(): void {
        console.log('Endpoint in task', this.auth.endpoint);
        this.rest.requestURLGet(this.auth.endpoint + 'tasks')
            .then(result => {
                this.task.taskList = this.persistence.jsonListToArray(result.objList);
                console.log(this.task.taskList);
                for (let x = 0; x < this.task.taskList.length; x++) {
                    this.rest.requestURLGet(this.task.taskList[x].links.find(obj => obj.rel === 'object').href)
                        .then(a => {
                            console.log('ObjectDetail', a);
                            this.task.taskList[x].links = a.links;
                        })
                        .catch(err => console.log(err));
                }
            });
    }

    /**
     * Adds the selected object to favorites
     *
     * @param object
     */
    addToFavorites(object: TIEObject): void {
        console.log('ADDED TO FAVS', object);
        this.rest.setAsFavorite(object.objId)
            .then( () => {
                this.toastCtrl.create({
                    message: object.attributes.objName + ' ' + this.translate.instant('TOAST.ADD_FAVORITE'),
                    duration: 3000,
                    position: 'bottom',
                }).present();
            });
    }

    switchView(): void {
        console.log('switching view');
        if (this.task.taskList.length <= 0) {
            this.toastCtrl.create({
                message: this.translate.instant('TOAST.NO_TASK'),
                duration: 3000,
                position: 'bottom',
            }).present();
        } else {
            console.log('Items Length', this.task.taskList.length);
            this.navCtrl.push(Object2FramesPage, {rel: 'SWITCHVIEW', payload: undefined}, {animate: true});
        }
    }

    openActionMenu(object: TIEObject): void {
        console.log('OPENACTION OBJECT', object);
        let modal = this.modalCtrl.create(OperationsModal, object);
        modal.present();
        modal.onWillDismiss(() => this.initTaskList());
    }

}

<ion-header>
    <ion-navbar>
        <ion-title>{{ 'TASK.TITLE' | translate }}</ion-title>
            <button ion-button menuToggle="left" start icon-only>
                <ion-icon name="menu"></ion-icon>
            </button>
        <ion-buttons end>
            <button ion-button icon-only (click)="switchView()">
                <ion-icon class="tie-one-finger-swipe-horizontally-icon"></ion-icon>

                <p>{{item.attributes.objId}}</p>
            </a>
            <ion-item-options>
                <button ion-button  color="secondary" (click)="addToFavorites(item)" icon-left>
                    <ion-icon class="tie-star-full-icon"></ion-icon>
