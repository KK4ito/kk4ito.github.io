import { Component } from '@angular/core';
import { NavParams, ViewController, NavController } from 'ionic-angular';
import { OperationService } from '../services/operations';
import { AttributesPage } from '../pages/attributes/attributes';
import { EditAttributesPage } from '../pages/attributes/editAttributes';
import { TIELink, TIEObject } from '../app/models/TIEObject';
import { OfflineService } from '../services/offline';
import { FileService } from '../services/file';
import { PersistenceService } from '../services/persistence';
import { DBOfflineObject } from '../app/models/DBObject';

@Component({
    templateUrl: 'operations.html',
})
export class OperationsModal {
    /**
     * TIEObject or DBOfflineObject
     */
    item: TIEObject | DBOfflineObject;
    availableOperations: Array<TIELink>;
    sharable: boolean;
    offlineCondition: boolean;

    constructor(private params: NavParams,
                private viewCtrl: ViewController,
                private operations: OperationService,
                private offline: OfflineService,
                private file: FileService,
                private persistence: PersistenceService,
                private navCtrl: NavController) {
        console.log('Enter: OperationsModal');
        console.log('Selected item for Modal', this.params.data);
        // assign item from params
        this.item = this.params.data;
        // is it an offlineItem
        this.offlineCondition = this.item.hasOwnProperty('localpath')
            && this.item.hasOwnProperty('owner')
            && this.item.hasOwnProperty('links');
        // filter to available operations
        this.availableOperations = this.getAvailableOperations();
        // if not offline check if it is sharable
        this.sharable = (this.offlineCondition) ? false : !!this.item.links.find(it => it.rel === 'VIEW');
    }

    getAvailableOperations(): Array<TIELink> {
        return this.persistence.filterLinks(this.item);
    }

    operationHandler(operation: TIELink): void {
        let link: string = operation.href;
        let name: string = operation.rel;
        switch (name) {
            case 'ATTRIBUTES_VIEW':
                this.navCtrl.push(AttributesPage, this.item);
                this.viewCtrl.dismiss({ rel: 'ATTRIBUTES_VIEW', payload: undefined});
                break;
            case 'BULK_ATTRIBUTES':
            case 'ATTRIBUTES_EDIT':
                this.operations.attributesEdit(link, this.item)
                    .then(output => {
                        this.navCtrl.push(EditAttributesPage, [output.object, output.urlPut]);
                        this.viewCtrl.dismiss({ rel: 'ATTRIBUTES_EDIT', payload: undefined});
                    });
                break;
            case 'VIEW':
                if (this.offlineCondition) {
                    this.file.open(link, (<DBOfflineObject>this.item).fileType);
                } else {
                    this.operations.view(link, this.item);
                }
                this.viewCtrl.dismiss({ rel: 'VIEW', payload: undefined});
                break;
            case 'CHECKIN':
                if (this.offlineCondition) {
                    this.offline.checkInOfflineFiles(<DBOfflineObject>this.item);
                } else {
                    this.operations.checkIn(link, this.item);
                }
                this.viewCtrl.dismiss({ rel: 'CHECKIN', payload: this.item});
                break;
            case 'CHECKOUT':
                this.operations.checkout(link, this.item);
                this.viewCtrl.dismiss({ rel: 'CHECKOUT', payload: undefined});
                break;
            case 'CONFIRM':
                this.operations.confirm(link);
                this.viewCtrl.dismiss({ rel: 'CONFIRM', payload: undefined});
                break;
            case 'CANCEL':
                this.operations.cancel(link);
                this.viewCtrl.dismiss({ rel: 'CANCEL', payload: undefined});
                break;
            case 'LIST':
                this.operations.list(link)
                    .then(output => this.viewCtrl.dismiss({ rel: 'LIST', payload: output}));
                break;
            case 'DELETE_OFFLINE_COPY':
                this.operations.deleteOfflineCopy((<TIEObject>this.item).objId)
                    .then(() => this.viewCtrl.dismiss({ rel: 'DELETE_OFFLINE_COPY', payload: this.item}));
                break;
            default:
        }
    }

    /**
     * Execute share (internal view and open)
     */
    generalShare() {
        this.operations.generalShare(this.item);
    }

    markOffline() {
        this.operations.markOffline(this.item);
    }

    addAsFavorite() {
        this.persistence.setAsFavorites(this.item)
            .then(() => this.viewCtrl.dismiss());
    }

    closeModal() {
        this.viewCtrl.dismiss({ rel: '', payload: undefined});
    }

}
