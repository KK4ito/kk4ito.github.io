import { Component } from '@angular/core';
import { ViewController, NavController, ToastController, MenuController, Platform } from 'ionic-angular';
import { AuthService } from '../../services/auth';
import { ErrorService } from '../../services/error';
import { OverviewPage } from '../overview/overview';
import { LoadingService } from '../../services/loading';
import { FileService } from '../../services/file';
import { NativeService } from '../../services/native';
import { TranslateService } from 'ng2-translate';
import { FingerPrintService } from '../../services/fingerprint';
import { PersistenceService } from '../../services/persistence';
import { TIEObject } from '../../app/models/TIEObject';

@Component({
    selector: 'ib-page-login',
    templateUrl: 'login.html',
})
export class LoginPage {

    model: {username: string, password: string, endpoint: string} = {username: '', password: '', endpoint: ''};
    touchIdAvailable: boolean = false;
    showEndpoint: boolean = false;

    constructor(private navCtrl: NavController,
                private viewCtrl: ViewController,
                private auth: AuthService,
                private toastCtrl: ToastController,
                private error: ErrorService,
                private file: FileService,
                private menu: MenuController,
                private persistence: PersistenceService,
                private platform: Platform,
                private load: LoadingService,
                private native: NativeService,
                private translate: TranslateService,
                private fingerprint: FingerPrintService) {
        console.log('Enter: LoginPage');
        this.fingerprint.isAvailable.subscribe(isAvailable => this.touchIdAvailable = isAvailable);
        this.auth.getEndpointFromStorage()
            .then(endpoint => {
                if (endpoint) {
                    this.model.endpoint = endpoint;
                } else {
                    this.model.endpoint = 'https://i-link.tie.ch/rest2/';
                }
            });
        this.check();
    }

    /**
     * Performs a check if a user is already logged in and redirects to overview.
     */
    check(): void {
        this.auth.authenticated()
            .then(status => (status) ? Promise.resolve('') : Promise.reject('Not logged in'))
            .then(() => this.auth.getUsername())
            .then(username => this.auth.username.next(username))
            .then(() => this.auth.setEndpointFromStorage())
            .then(() => this.persistence.getCurrent())
            .then(() => {
                this.file.initUserFolder();
                this.auth.isLoggedIn = true;
                this.navCtrl.setRoot(OverviewPage, {}, {animate: false});
            })
            .catch(err => console.error('check Error: ', err));
    }

    /**
     * Check if the user has logged out or expired.
     *
     * @returns boolean
     */
    ionViewCanEnter() {
        return !this.auth.authenticatedImmediate();
    }

    /**
     * Prepare data of the component
     */
    ionViewWillEnter() {
        this.menu.enable(false, 'mainMenu');
        this.menu.enable(false, 'rightMenu');
        if (this.persistence.schemeObserver.observers === null) this.persistence.schemeObserver.unsubscribe();
        this.native.registerBackButton(() => this.platform.exitApp());
    }

    /**
     * Reenable sidemenu
     */
    ionViewWillLeave() {
        this.menu.enable(true);
        this.menu.enable(true, 'rightMenu');
        this.viewCtrl.showBackButton(false);
    }

    /**
     * Login handler. Shows error modal if not authenticated.
     */
    login(): void {
        this.load.loading(true, this.translate.instant('LOGIN.LOG_IN'));
        this.model.username = this.model.username.toLowerCase();
        this.persistence.initDB()
            .then(() => {
                this.auth.setCredentials(this.model).then(() => {
                    this.persistence.getEntryPoint().then(getEntry => {
                        this.file.initUserFolder();
                        this.persistence.createUser(this.model.username)
                            .then(() => this.persistence.applyUserConfig());
                        this.persistence.getObjectFromServer(
                            getEntry.objList[Object.keys(getEntry.objList)[0]].links
                                .find(l => l.rel === 'FIND').href, true)
                            .then(putEntry => {
                                let favoriteObject: TIEObject = JSON.parse(JSON.stringify(putEntry));
                                let merge = Object.assign({}, putEntry);
                                let firstGetObject = getEntry.objList[Object.keys(getEntry.objList)[0]];
                                merge.links = Object.assign(merge.links,
                                    firstGetObject.links.concat(favoriteObject.links));
                                console.log('mergeObject', merge);
                                this.auth.isLoggedIn = true;
                                this.persistence.createObjectsInDB([merge], 0, 1)
                                    .then(() => this.persistence.setCurrent(merge, 0))
                                    .then(() => this.navCtrl.push(OverviewPage, {}, {animate: false}))
                                    .then(() => this.load.loading(false))
                                    .catch(err => console.error('Login failed', err,
                                        putEntry, this.auth.username.value));
                            });
                    }).catch(err => {
                        this.auth.logout();
                        this.load.loading(false);
                        this.error.handleHttpError(err, 'LoginPage:login');
                    });
                });
            });
    }

    /**
     * Form validator. Validates before the submit fires the login() function.
     *
     * @param formItems submitted && not valid form elements
     */
    validate(formItems: boolean): void {
        if (formItems) {
            let toast = this.toastCtrl.create({
                message: this.translate.instant('LOGIN.REQUIRED'),
                duration: 3000,
                position: 'bottom',
            });

            toast.present();
        }
    }

    startTouchID() {
        this.model.password = '';
        this.fingerprint.verifyFingerPrint(this.model.username)
            .then(decrypted => {
                this.model.password = decrypted;
                this.login();
            });
    }
}
