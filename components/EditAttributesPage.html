import { NavController } from 'ionic-angular';
import { Component } from '@angular/core';
import { RestService } from '../../services/rest';
import { NavParams, ToastController } from 'ionic-angular';
import { ErrorService } from '../../services/error';
import { NativeService } from '../../services/native';
import { TranslateService } from 'ng2-translate';
import { PersistenceService } from '../../services/persistence';

@Component({
    templateUrl: 'editAttributes.html',
})
export class EditAttributesPage {
    items;
    linkToPut;

    constructor(private params: NavParams,
                private navCtrl: NavController,
                private rest: RestService,
                private error: ErrorService,
                private native: NativeService,
                private translate: TranslateService,
                private persistence: PersistenceService,
                private toastCtrl: ToastController) {
        this.items = [];
        let attributes = this.params.data[0];
        this.linkToPut = this.params.data[1];
        for (let properties in attributes) {
            if (attributes.hasOwnProperty(properties)) {
                let obj = attributes[properties];
                this.items.push({key: properties, value: obj});
            }
        }
    }

    ionViewWillEnter() {
        this.native.registerBackButton(() => {
            this.back();
        });
    }

    /**
     * TODO: update object and attributes in DB.
     * Saves changes to server and DB.
     */
    saveChanges(): void {
        let temp = {};
        for (let i = 0; i < this.items.length; i++) {
            if (this.items[i] !== null || this.items[i].key !== null) {
                temp[this.items[i].key] = this.items[i].value.value;
            }
        }
        let changes = '{"attributes": ' + JSON.stringify(temp) + '}';

        // TODO Validating
        this.rest.requestURLPut(this.linkToPut, JSON.parse(changes))
            .then(result => {
                this.toastCtrl.create({
                    message: this.translate.instant('TOAST.SAVED_CHANGES'),
                    duration: 3000,
                    position: 'bottom',
                }).present();
                console.log('getACtive', this.navCtrl);
                console.log('getPrevious', this.navCtrl);
                if (this.navCtrl.getPrevious().name === 'OverviewPage') {
                    this.persistence.addNext(result);
                    this.persistence.updateSingleObjectInCache(result, this.persistence.current.value.id);
                }
                let newList = this.persistence.items.value.map(t => (t.objId === result.objId) ? result : t);
                this.persistence.items.next(newList);
            })
            .catch(err => {
                console.log(err);
                this.error.handleHttpError(err, 'EditAttributesPage:saveChanges');
            });
    }

    /**
     *
     */
    back() {
        this.navCtrl.pop();
    }
}
