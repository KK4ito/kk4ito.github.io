import { Component, ViewChild } from '@angular/core';
import { FormControl } from '@angular/forms';
import { Platform, ModalController, Nav, App, MenuController, Content } from 'ionic-angular';
import { StatusBar, Splashscreen } from 'ionic-native';
import { LoginPage } from '../pages/login/login';
import { OverviewPage } from '../pages/overview/overview';
import { SettingsPage } from '../pages/settings/settings';
import { OfflinePage } from '../pages/offline/offline';
import { AuthService } from '../services/auth';
import { SearchService } from '../services/search';
import { RestService } from '../services/rest';
import { ErrorService } from '../services/error';
import { NetworkService } from '../services/network';
import { TIEObject } from './models/TIEObject';
import { OperationsModal } from '../modals/operations';
import { OperationService } from '../services/operations';
import { TranslateService, TranslationChangeEvent } from 'ng2-translate';
import { DBService } from '../services/db';
import { OfflineService } from '../services/offline';
import { TaskPage } from '../pages/task/task';
import { PersistenceService } from '../services/persistence';
import { PushService } from '../services/push';

declare let document;

@Component({
    templateUrl: `app.html`,
})
export class MyApp {
    @ViewChild(Nav) nav: Nav;
    @ViewChild(Content) content: Content;
    rootPage: any;
    pages: Array<{title: any, icon: string, component: any, active: boolean}>;
    username: string;
    searchValue: FormControl;
    searchResults: Array<TIEObject>;
    offlineState: boolean;
    isLoading: boolean;
    hasSearched: boolean;
    snapShotValue: string;

    constructor(public platform: Platform,
                public auth: AuthService,
                public app: App,
                public error: ErrorService,
                public networkService: NetworkService,
                public searchService: SearchService,
                public modalCtrl: ModalController,
                public menuCtrl: MenuController,
                public operation: OperationService,
                public rest: RestService,
                public translate: TranslateService,
                public db: DBService,
                public persistence: PersistenceService,
                public offline: OfflineService,
                public push: PushService) {
        this.initializeApp();
        this.rootPage = LoginPage;
        this.pages = [
            { title: 'Explorer', icon: 'tie-list-icon', component: OverviewPage, active: true },
            { title: 'Tasks', icon: 'tie-list-numbered-icon', component: TaskPage, active: false },
            { title: 'Einstellungen', icon: 'tie-cog-icon', component: SettingsPage, active: false },
            { title: 'Offline Dateien', icon: 'tie-list-icon', component: OfflinePage, active: false },
        ];
        this.username = '';
        this.searchValue = new FormControl();
        this.searchResults = [];
        this.offlineState = false;
        this.isLoading = false;
        this.hasSearched = false;
        this.snapShotValue = '';
    }

    executeSearch(): void {
        this.isLoading = true;
        this.hasSearched = true;
        this.searchResults = [];
        this.snapShotValue = this.searchValue.value;
        this.searchService.search(this.searchValue.value)
            .then(result => {
                this.searchResults = result;
                this.isLoading = false;
            });
    }

    /**
     * Initialize the app
     */
    initializeApp(): void {
        this.platform.ready().then(() => {
            this.push.init();
            this.translate.setDefaultLang('en');
            StatusBar.backgroundColorByHexString('#a72e29');
            Splashscreen.hide();
            this.persistence.initDB()
                .then(() => {
                    this.translate.onDefaultLangChange.subscribe((event: TranslationChangeEvent) => {
                        this.pages[0].title = event.translations.OVERVIEW.TITLE;
                        this.pages[1].title = event.translations.TASKS.TITLE;
                        this.pages[2].title = event.translations.SETTINGS.TITLE;
                        this.pages[3].title = event.translations.OFFLINE.TITLE;
                    });
                    this.networkService.isOfflineObserver.subscribe((value) => {
                        console.log('Observable triggered: network', value);
                        this.offlineState = value;
                    });
                    this.persistence.schemeObserver.subscribe((scheme) => {
                        this.persistence.updateColorScheme(scheme);
                    });
                    this.auth.username.subscribe((username) => {
                        if (username) {
                            this.username = username.toUpperCase();
                            this.persistence.getUserConfig(username)
                                .then(config => {
                                    console.log('getUserConfig', config);
                                    this.translate.setDefaultLang(config.find(e => e.I18N === 'LANGUAGE').VALUE);
                                    this.persistence.schemeObserver.next(config.find(e => e.I18N === 'COLOR_SCHEME').VALUE);
                                })
                                .catch(err => console.error(err));
                        }
                    });
                });
        });
    }

    /**
     * Opens the selected page component.
     *
     * @param page
     */
    openPage(page): void {
        // Reset the content nav to have just this page
        // we wouldn't want the back button to show in this scenario
        this.nav.setRoot(page.component);
        this.pages.map(it => it.active = false);
        this.pages.map(it => (page.title === it.title) ? it.active = true : '');
    }

    /**
     * Logout the user
     */
    logout(): void {
        this.auth.logout();
        this.nav.setRoot(LoginPage);
    }

    /**
     * TODO: evtl woanderse platzieren
     * Create and present modal for the clicked search result
     * @param object clicked search Result
     */
    openSearchResult(object: TIEObject) {
        this.rest.requestURLGet(object.links.find(obj => obj.rel === 'object').href)
            .then( (newObject: TIEObject) => {
                newObject.links = this.persistence.filterLinks(newObject);
                console.log('clicked object and result of filterlinks', newObject);
                let modal = this.modalCtrl.create(OperationsModal, newObject);
                modal.onWillDismiss(data => {
                    if (data.payload) {
                        console.log('dismiss data', data);
                    } else {
                        console.log('Back in overview is not list operation');
                    }
                });
                this.menuCtrl.close('rightMenu');
                this.menuCtrl.close('mainMenu');
                modal.present();
            });
    }
}
