import { Component } from '@angular/core';
import { OfflineService } from '../../services/offline';
import { DBOfflineObject } from '../../app/models/DBObject';
import { OperationsModal } from '../../modals/operations';
import { ModalController } from 'ionic-angular';
import { PersistenceService } from '../../services/persistence';
import { TIEObject } from '../../app/models/TIEObject';

@Component({
    selector: 'ib-page-settings',
    templateUrl: 'offline.html',
})
export class OfflinePage {

    items: Array<DBOfflineObject>;

    constructor(private offline: OfflineService,
                private modalCtrl: ModalController,
                private persistence: PersistenceService) {
        console.log('Enter: OfflinePage');
        this.items = [];
    }

    ionViewWillEnter() {
        this.refreshOfflineList();
    }

    /**
     * Opens a modal and shows operations on a checked out file in offline mode.
     *
     * @param object
     */
    openActionMenu(object: DBOfflineObject): void {
        console.log('entering openActionmenu offline/offline', object);
        let obj: TIEObject = JSON.parse(object.data);
        let tempCheckin = obj.links.find(it => it.rel === 'CHECKIN');
        object.attributes = { objName: obj.attributes.objName };
        object.links = [];
        object.links.push({rel: 'VIEW', href: object.localpath});
        if (tempCheckin) {
            object.links.push({rel: 'CHECKIN', href: tempCheckin.href});
        } else {
            object.links.push({rel: 'DELETE_OFFLINE_COPY', href: object.offlineId.toString()});
        }
        let modal = this.modalCtrl.create(OperationsModal, object);
        modal.onWillDismiss((item) => {
            if ((item.rel === 'CHECKIN' || item.rel === 'DELETE_OFFLINE_COPY') &&
                item.payload) this.items.splice(this.items.indexOf(item), 1);
        });
        modal.present();
    }
    /**
     * Refresh the list with offline files.
     */
    refreshOfflineList() {
        this.offline.getOfflineFiles()
            .then(result => {
                console.log('result of db read', result);
                if (result) this.items = (Array.isArray(result)) ? result : [result];
            });
    }
}

<ion-header>
    <ion-navbar>
        <button ion-button menuToggle>

        </button>

    </ion-navbar>

<ion-content>
    <ion-item *ngFor="let item of items">
        <a ion-item
           (press)="openActionMenu(item)">
            <h2>{{item.localpath}}</h2>
            <h2>{{item.offlineId}}</h2>
            <p>{{item.owner}}</p>
            <p>{{item.localpath.split('/').pop()}}</p>
        </a>
    </ion-item>
    <ion-item *ngIf="items.length == 0" text-center text>
        {{ 'OFFLINE.NO_FILES' | translate }}
    </ion-item>
