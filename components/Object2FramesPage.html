import { Component, ViewChild, ViewChildren, QueryList } from '@angular/core';
import 'rxjs/Rx';
import {
    StackConfig,
    DragEvent,
    SwingStackComponent,
    SwingCardComponent } from 'angular2-swing';
import { TIEObject } from '../../app/models/TIEObject';
import { OperationService } from '../../services/operations';
import { TaskService } from '../../services/task';
import { ToastController } from 'ionic-angular';
import { TranslateService } from 'ng2-translate';

@Component({
    templateUrl: 'object2Frames.html',
})
export class Object2FramesPage {
    @ViewChild('myswing1') swingStack: SwingStackComponent;
    @ViewChildren('mycards1') swingCards: QueryList<SwingCardComponent>;

    cards: Array<TIEObject>;
    stackConfig: StackConfig;

    constructor(private operations: OperationService,
                private task: TaskService,
                private translate: TranslateService,
                private toastCtrl: ToastController) {
        this.cards = [];
        this.stackConfig = {
            throwOutConfidence: (offsetX, _, element) => {
                return Math.min(Math.abs(offsetX) / (element.offsetWidth / 2), 1);
            },
            transform: (element, x, y, r) => {
                this.onItemMove(element, x, y, r);
            },
            throwOutDistance: () => {
                return 800;
            },
        };
    }

    /**
     * Behaviour different if coming from Task view or back from an inapp/file view
     */
    ionViewDidLoad() {
        console.log('load o2f');
        // Either subscribe in controller or set in HTML
        this.swingStack.throwin.subscribe((event: DragEvent) => {
            event.target.style.background = '#ffffff';
        });
        this.addNextCards(true);
    }

    /**
     * Translation method for cards
     *
     * @param element
     * @param x
     * @param y
     * @param r
     */
    onItemMove(element, x, y, r) {
        let color = '';
        let abs = Math.abs(x);
        let min = Math.trunc(Math.min(16 * 16 - abs, 16 * 16));
        let hexCode = this.decimalToHex(min, 2);

        if (x < 0) {
            color = '#FF' + hexCode + hexCode;
        } else {
            color = '#' + hexCode + 'FF' + hexCode;
        }

        element.style.background = color;
        element.style['transform'] = `translate3d(0, 0, 0) translate(${x}px, ${y}px) rotate(${r}deg)`;
    }

    /**
     * Voting method for cards
     *
     * @param like
     */
    voteUp(like: boolean) {
        // pop card from this.cards
        let removedCard: TIEObject = this.cards.pop();
        if (this.task.taskList.length > 0) this.addNextCards(true);
        // server request for this task
        if (like) {
            this.operations.confirm(removedCard.links.find(l => l.rel === 'CONFIRM').href)
                .then(() => {
                    this.toastCtrl.create({
                        message: this.translate.instant('TOAST.LIKED', removedCard.attributes),
                        duration: 3000,
                        position: 'bottom',
                    }).present();
                });
        } else {
            this.operations.cancel(removedCard.links.find(l => l.rel === 'CANCEL').href)
                .then(() => {
                    this.toastCtrl.create({
                        message: this.translate.instant('TOAST.DISLIKED', removedCard.attributes),
                        duration: 3000,
                        position: 'bottom',
                    }).present();
                });
        }
    }

    /**
     * Adds card as currentTask and try to open file if allowOpening is true
     *
     * @param allowOpening
     */
    addNextCards(allowOpening: boolean) {
        let item = this.task.taskList.pop();
        if (item) this.cards.push(item);
        if (item && allowOpening) {
            let view = item.links.find(l => l.rel === 'VIEW');
            if (view) {
                this.operations.view(view.href, item);
            } else if (item.attributes.objName && item.attributes.descr) {
                console.log('No VIEW Link but objName and desc available');
            } else {
                let link = item.links.find(l => l.rel === 'ATTRIBUTES_VIEW');
                if (link) this.operations.attributesView(link.href);
            }
        }
    }

    /**
     * http://stackoverflow.com/questions/57803/how-to-convert-decimal-to-hex-in-javascript
     */
    decimalToHex(d, padding) {
        let hex = Number(d).toString(16);
        padding = typeof (padding) === 'undefined' || padding === null ? padding = 2 : padding;

        while (hex.length < padding) {
            hex = '0' + hex;
        }

        return hex;
    }
}
