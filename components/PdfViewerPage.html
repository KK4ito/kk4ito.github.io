import { Component } from '@angular/core';
import { NavParams, MenuController, NavController } from 'ionic-angular';
import { NativeService } from '../../services/native';

@Component({
    templateUrl: 'pdfViewer.html',
})
export class PdfViewerPage {
    pdfSource: string;
    fileName: string;
    pageVariable: number;
    maxPages: number;

    constructor(private params: NavParams,
                private menuCtrl: MenuController,
                private native: NativeService,
                private navCtrl: NavController) {
        console.log('Enter: PDFViewer');
        console.log(params);
        this.pdfSource = params.data[0].fullPath;
        this.fileName = params.data[1].attributes.objName + params.data[1].attributes.objclassDescr;
        this.pageVariable = 1;
    }

    ionViewWillEnter() {
        this.menuCtrl.swipeEnable(false);
        this.menuCtrl.swipeEnable(false, 'rightMenu');
        this.native.registerBackButton(() => this.navCtrl.pop());
    }

    ionViewWillLeave() {
        this.menuCtrl.swipeEnable(true);
        this.menuCtrl.swipeEnable(true, 'rightMenu');

    }

    pdfLoaded(event: PDFDocumentProxy) {
        this.maxPages = event.numPages;
    }

    increment() {
        if (this.pageVariable < this.maxPages) ++this.pageVariable;
    }

    decrement() {
        if (this.pageVariable > 1) --this.pageVariable;
    }

    swipeEvent(e) {
        if (e.deltaX < 0) { // Swipe from right to left
            console.log('Swipe from right to left', e);
        } else { // Swipe from left to right
            console.log('Swipe from left to right', e);
        }
    }

}

<ion-header>
    <ion-navbar>

            <span class="pdf-page-count">{{pageVariable}}/{{maxPages}}</span>

    <ion-grid class="button-group">
