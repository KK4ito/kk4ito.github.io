import { Component, ViewChild } from '@angular/core';
import { FileChooser, Camera, Transfer, FileUploadResult, FileUploadOptions } from 'ionic-native';
import {
    App, Platform, MenuController, ModalController, ViewController,
    AlertController, ToastController, Content,
}
    from 'ionic-angular';
import { AuthService } from '../../services/auth';
import { OperationsModal } from '../../modals/operations';
import { OperationService } from '../../services/operations';
import { NetworkService } from '../../services/network';
import { LoadingService } from '../../services/loading';
import { ErrorService } from '../../services/error';
import { TIEObject } from '../../app/models/TIEObject';
import { DBObject } from '../../app/models/DBObject';
import * as _ from '../../app/consts/objclassConsts.json';
import { TranslateService } from 'ng2-translate';
import { Subscription } from 'rxjs';
import { PersistenceService } from '../../services/persistence';

@Component({
    selector: 'ib-page-overview',
    templateUrl: 'overview.html',
})
export class OverviewPage {
    @ViewChild(Content) content: Content;
    items: Array<TIEObject>;
    current: DBObject;
    itemsSubscription: Subscription;
    currentSubscription: Subscription;
    hasPrevious: boolean;
    hasDragDrop: boolean;
    canLoadMore: boolean;
    lastBack = 0;
    offlineState: boolean;

    constructor(private app: App,
                private platform: Platform,
                private modalCtrl: ModalController,
                private auth: AuthService,
                private networkService: NetworkService,
                private viewCtrl: ViewController,
                private alertCtrl: AlertController,
                private toastCtrl: ToastController,
                private menuCtrl: MenuController,
                private operation: OperationService,
                private load: LoadingService,
                private persistence: PersistenceService,
                private error: ErrorService,
                private translate: TranslateService) {
        console.log('Enter: OverviewPage');
        this.current = null;
        this.hasDragDrop = false;
        this.canLoadMore = false;
        this.hasPrevious = false;
    }

    /**
     * Check if we are logged in to enter this component
     *
     * @returns boolean
     */
    ionViewCanEnter(): boolean {
        return this.auth.authenticatedImmediate();
    }

    /**
     * Enable sideMenus
     */
    ionViewDidEnter(): void {
        this.menuCtrl.enable(true, 'mainMenu');
        this.menuCtrl.enable(true, 'rightMenu');
    }

    ionViewWillEnter() {
        this.itemsSubscription = this.persistence.items.subscribe(items => {
            this.items = items;
        });
        this.currentSubscription = this.persistence.current.subscribe(current => {
            this.content.scrollToTop();
            console.log('BEFORE PARSING JSON currnet,data', current);
            let object = Object.assign({}, JSON.parse(current.data));
            this.current = Object.assign({}, current);
            this.current.data = JSON.parse(this.current.data);
            this.checkForDragAndDrop();
            this.checkIfHasPrevious();
            this.items = this.persistence.jsonListToArray(object.objList);
            this.canLoadMore = this.items.length >= 10;
        });
        this.platform.registerBackButtonAction(() => {
            let modal = this.app._appRoot._modalPortal.getActive();
            const closeDelay = 2000;
            console.log('registerBackButtonAction', this.hasPrevious);
            if (modal && modal.dismiss) {
                modal.dismiss();
            } else if (this.menuCtrl.isOpen()) {
                this.menuCtrl.close();
            } else if (this.hasPrevious) {
                this.goUp();
            } else if (Date.now() - this.lastBack < closeDelay) {
                this.presentConfirm();
            }
            this.lastBack = Date.now();
        });
    }

    ionViewDidLoad(): void {
        console.log('DIDLOAD LIFECYLCE from overview');
        if (!this.networkService.isOffline) {
            this.viewCtrl.showBackButton(false);
            this.offlineState = false;
            this.init();
        } else {
            this.offlineState = true;
            console.log('Cancelled init from overview');
        }
    }

    ionViewDidLeave(): void {
        this.currentSubscription.unsubscribe();
        this.itemsSubscription.unsubscribe();
    }

    goOnline() {
        console.log('goOnline triggerd');
        if (!this.networkService.isOffline) {
            this.offlineState = false;
            this.viewCtrl.showBackButton(false);
            this.init();
        } else {
            this.offlineState = true;
            let alert = this.alertCtrl.create({
                title: this.translate.instant('ALERT.CONNECTION.TITLE'),
                subTitle: this.translate.instant('ALERT.CONNECTION.DESCRIPTION'),
                buttons: [this.translate.instant('ALERT.CLOSE')],
            });
            alert.onDidDismiss( () => {
                if (!this.networkService.isOffline) {
                    this.offlineState = false;
                    this.init();
                }
            });
            alert.present();
        }
    }
    extractObjectClassID(item: TIEObject) {
        return this.persistence.extractObjectClassID(item);
    }

    checkForDragAndDrop(): void {
        this.hasDragDrop = !!(<TIEObject>this.current.data).links.find(l => l.rel === 'DRAGDROP');
    }

    presentConfirm() {
        this.alertCtrl.create({
            title: 'Exit app',
            message: 'Do you want to close the app?',
            buttons: [
                {
                    text: 'No',
                    role: 'cancel',
                    handler: () => {},
                },
                {
                    text: 'Yes',
                    handler: () => this.platform.exitApp(),
                },
            ],
        }).present();
    }
    checkIfHasPrevious(): void {
        this.hasPrevious = this.current.parent !== 0;
    }

    /**
     * Initializer for entry point. Is called only once in constructor
     */
    init(): void {
        this.persistence.getCurrent()
            .then((current: DBObject) => {
                let currentObject = JSON.parse(current.data);
                let currentObjList = this.persistence.jsonListToArray(currentObject.objList);
                this.persistence.addNext(currentObject);
                this.persistence.items.next(currentObjList);
                this.persistence.cacheFileObjects(currentObject, currentObjList);
            });
    }

    /**
     * Executes camera or file process.
     *
     * @param fab
     * @param camera
     */
    createAction(fab: any, camera?: boolean): void {
        let currentObject: TIEObject = this.current.data;
        if (this.current.type === _.Register) {
            let url = currentObject.links.find(it => it.rel === 'DRAGDROP').href;
            if (camera) {
                Camera.getPicture()
                    .then( imageData => {
                        this.uploadFileWithDialog(imageData, fab, url);
                    });
            } else {
                FileChooser.open()
                    .then(uri => {
                        this.uploadFileWithDialog(uri, fab, url);
                    });
            }
        }
    }

    /**
     * Check if it is a file and if a checkout link is available
     *
     * @param object
     * @returns {string}
     */
    objectStatusChecker(object: TIEObject): string {
        let type = this.extractObjectClassID(object);
        // default color
        let color = 'green';
        // if dossier/register black
        if (type === _.Dossier || type === _.Register) color = 'black';
        // if file is checked out
        if (object.attributes.checkedOutBy) color = 'red';

        return color;
    }

    /**
     * Opens a modal to interact with an object.
     * This shows all links (operations) on an object.
     *
     * @param object
     */
    openActionMenu(object: TIEObject): void {
        // show filtered links in modal
        let type = this.extractObjectClassID(object);
        new Promise((resolve, reject) => {
            if (type === _.Dossier || type === _.Register) {
                this.persistence.getObjectFromDB(object)
                    .then((objFromDB: DBObject) => resolve(JSON.parse(objFromDB.data)))
                    .catch(err => reject(err));
            } else {
                let objFromCache = this.persistence.getFromCache(this.current.id, object.objId);
                console.log('Object From Cache', objFromCache);
                if (objFromCache) {
                    resolve(objFromCache);
                } else {
                    console.log('no obj found in cache', this.persistence.objectCache, this.current.id, object.objId);
                    reject();
                }
            }
        })
            .then(obj => {
                let modal = this.modalCtrl.create(OperationsModal, obj);
                modal.onWillDismiss(data => (data.rel === 'LIST') ? this.goDown(object) : '');
                modal.present();
            })
            .catch(err => console.log('Error during openActionMenu: ', err));
    }
    /**
     * Decides which immediate action to execute
     *
     * @param object
     */
    executeAction(object: TIEObject): void {
        let type = this.extractObjectClassID(object);
        // if dossier/register
        switch (type) {
            case _.Dossier:
            case _.Register:
                this.goDown(object);
                break;
            default:
                let objCacheElement = <TIEObject>this.persistence.getFromCache(this.current.id, object.objId);
                let view = objCacheElement.links.find(l => l.rel === 'VIEW');
                if (view) {
                    this.operation.view(view.href, objCacheElement);
                } else {
                    let link = objCacheElement.links.find(l => l.rel === 'ATTRIBUTES_VIEW');
                    if (link) this.operation.attributesView(link.href);
                }
        }
    }

    /**
     * Navigate one level down into an object
     *
     * @param object
     */
    goDown(object: TIEObject): void {
        let oldCurrent = Object.assign({}, this.current);
        this.persistence.getObjectFromDB(object)
            .then((obj: DBObject) => {
                let parsedObj: TIEObject = JSON.parse(obj.data);
                this.persistence.addNext(parsedObj);
                this.persistence.updatePrevious(oldCurrent.data);
                console.log('goDown before setCurrent', parsedObj, oldCurrent);
                this.persistence.setCurrent(parsedObj, oldCurrent.id)
                    .then(() => {
                        let list = this.persistence.jsonListToArray(parsedObj.objList);
                        this.persistence.items.next(list);
                        return list;
                    })
                    .then(items => this.persistence.cacheFileObjects(parsedObj, items))
                    .catch(err => console.error('goDown Error: ', err));
            });
    }

    /**
     * Navigate back up a level in the objects list
     */
    goUp(): void {
        let parentId = this.current.parent;
        this.persistence.getObjectFromDB({objId: parentId})
            .then((parent: DBObject) => {
                this.checkIfHasPrevious();
                let parentParsed = JSON.parse(parent.data);
                let list = this.persistence.jsonListToArray(parentParsed.objList);
                this.persistence.getObjectFromDB({objId: parent.parent})
                    .then((parentParent: DBObject) => {
                        if (parent.parent !== 0) {
                            let parentParentParsed = JSON.parse(parentParent.data);
                            console.log('goUp parent not 0', parentParent);
                            this.persistence.updatePrevious(parentParentParsed);
                            this.persistence.cacheFileObjects(parentParentParsed, list);
                        } else {
                            this.persistence.cacheFileObjects({
                                objId: parent.id,
                                objList: parentParsed.objList,
                            }, list);
                        }
                    })
                    .catch(err => console.error('inside goUp() Error: ', err));
                this.persistence.addNext(parentParsed);
                this.persistence.setCurrent(parentParsed, parent.parent)
                    .then(() => {
                        this.persistence.items.next(list);
                    });
            })
            .catch(err => console.error('goup Error: ', err));
    }

    /**
     * Adds the selected object to favorites
     *
     * @param object
     */
    addToFavorites(object: TIEObject): void {
        this.persistence.setAsFavorites(object);
    }

    /**
     * Removes the selected object from favorites
     *
     * @param object
     */
    removeFromFavorites(object: TIEObject): void {
        this.persistence.removeFromFavorite(object.objId);
        let toast = this.toastCtrl.create({
            message: object.attributes.objName + ' von Favoriten entfernt',
            duration: 3000,
            position: 'bottom',
        });
        // Remove favorites from the current list
        this.items.splice(this.items.indexOf(object), 1);
        this.persistence.items.next(this.items);
        toast.present();
    }

    /**
     * Load the next batch of items, since the current view only supports 10 items
     * @param infiniteScroll
     */
    loadMore(): Promise<any> {
        return new Promise((resolve) => {
            let temp = (<TIEObject>this.current.data).links.find(l => l.rel === 'next');
            let putUrl = (<TIEObject>this.current.data).links.find(l => l.rel === 'FIND');
            if (temp) {
                this.persistence.getObjectFromServer(temp.href, putUrl !== undefined)
                    .then(res => {
                        // Get new objects from next url
                        let newObjectList = this.persistence.jsonListToArray(res.objList);
                        this.persistence.addNext(res);
                        // Check if another next url exists
                        let nextLink = res.links.find(it => it.rel === 'next');
                        if (nextLink) {
                            this.canLoadMore = true;
                            (<TIEObject>this.current.data).links.find(l => l.rel === 'next').href = nextLink.href;
                        } else {
                            this.canLoadMore = false;
                        }
                        // Refresh List of items and store (if object) in cache
                        this.persistence.cacheFileObjects(res, newObjectList)
                            .then(() => {
                                this.persistence.items.next(this.items.concat(newObjectList));
                                resolve();
                            });
                    });
            } else {
                resolve();
            }
        });
    }

    /**
     * Upload a new file
     *
     * @param filePath
     * @param fab Floating Action Button (Html DOM object)
     * @param locationToUpload
     */
    uploadFileWithDialog(filePath: string, fab: any, locationToUpload: string): void {
        let prompt = this.alertCtrl.create({
            title: this.translate.instant('UPLOAD.TITLE'),
            message: this.translate.instant('UPLOAD.MESSAGE'),
            inputs: [
                {
                    name: 'objName',
                    placeholder: this.translate.instant('UPLOAD.INPUTS.OBJNAME'),
                },
                {
                    name: 'descr',
                    placeholder: this.translate.instant('UPLOAD.INPUTS.OBJDESC'),
                },
            ],
            buttons: [
                {
                    text: 'Cancel',
                },
                {
                    text: 'Save',
                    handler: data => {
                        if (data.objName) {
                            console.log('selectede file for upload', data);
                            this.load.loading(true, this.translate.instant('UPLOAD.UPLOADING'));
                            let fileName = filePath.substr(filePath.lastIndexOf('/') + 1, filePath.length);
                            this.persistence.getObjclassIdFromURI(filePath, locationToUpload)
                                .then(objclassId => {
                                    console.log('objclassId', objclassId);
                                    let transfer = new Transfer();
                                    let options: FileUploadOptions = {
                                        fileKey: 'file',
                                        fileName: fileName,
                                        params: {
                                            objclassId: objclassId,
                                            attrs: JSON.stringify({
                                                attributes: {
                                                    objName: data.objName,
                                                    descr: data.descr,
                                                }}),
                                        },
                                    };
                                    transfer.upload(filePath, locationToUpload, options, true)
                                        .then((result: FileUploadResult) => {
                                            // Autoupdate View with the new item
                                            let newItem: TIEObject = JSON.parse(result.response);
                                            this.toastCtrl.create({
                                                message: this.translate.instant('UPLOAD.SUCCESS'),
                                                duration: 3000,
                                                position: 'bottom',
                                            }).present();
                                            // add item to the list if not more than 9 items in it
                                            if (this.items.length <= 9)
                                                this.persistence.items.next(this.items.concat([newItem]));
                                            // update previous
                                            this.persistence.updatePrevious({objId: this.current.id});
                                            // add item into cache
                                            this.persistence.cacheFileObjects(this.current.data, [newItem]);
                                            this.load.loading(false);

                                        })
                                        .catch(err => {
                                            this.load.loading(false);
                                            this.error.handleFileError(err, this.translate.instant('UPLOAD.ERROR'));
                                        });
                                });
                        } else {
                            this.uploadFileWithDialog(filePath, fab, locationToUpload);
                            this.error.handleCustomError(this.translate.instant('UPLOAD.MISSING'));
                        }
                    },
                },
            ],
        });
        prompt.present();
        fab.close();
    }
}
