import { Component } from '@angular/core';
import { NavParams, MenuController, NavController, Platform } from 'ionic-angular';
import { NativeService } from '../../services/native';

declare let AndroidFullScreen;

@Component({
    templateUrl: 'videoViewer.html',
})
export class VideoViewerPage {
    fileName: string;
    videoSource: string;
    fileType: string;

    constructor(private params: NavParams,
                private menuCtrl: MenuController,
                private native: NativeService,
                private navCtrl: NavController,
                private platform: Platform) {
        console.log('Enter: VideoViewer');
        if (this.platform.is('ios')) {
            this.fileType = 'video/mp4';
            this.videoSource = params.data[0].fullPath;
            this.fileName = params.data[1].attributes.objName + params.data[1].attributes.objclassDescr;
        } else {
            let temp = params.data.links.find(it => it.rel === 'STREAM');
            this.videoSource = temp.href;
            this.fileType = 'video/mp4';
            this.fileName = params.data.attributes.objName + params.data.attributes.objclassDescr;
        }
    }

    ionViewWillEnter() {
        this.menuCtrl.swipeEnable(false);
        this.menuCtrl.swipeEnable(false, 'rightMenu');
        this.native.registerBackButton(() => this.navCtrl.pop());
        if (this.platform.is('android')) {
            AndroidFullScreen.immersiveMode(() => {}, () => {});
        }
    }

    ionViewWillLeave() {
        this.menuCtrl.swipeEnable(true);
        this.menuCtrl.swipeEnable(true, 'rightMenu');
        if (this.platform.is('android')) {
            AndroidFullScreen.showSystemUI(() => {}, () => {});
        }
    }
}
